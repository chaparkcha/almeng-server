```
npm init

npm install express // 웹 서버 모듈

npm i --save-dev @babel/core @babel/cli @babel/preset-env // es6 문법을 브라우저에서 동작하도록 해주는 트랜스파일러

npm install @babel/polyfill // es6+ 에서 추가된 메소드를 사용할수 있게 해준다.

npm install --save-dev webpack webpack-cli // js 파일 번들러

npm install --save-dev babel-loader // 웹펙에서 babel을 사용하기위해 로더 설정

npm install node-polyfill-webpack-plugin // 웹팩 5에서 nodejs의 필수모듈을 지원하지 않아서 추가로 설치한 플러그인모음
```
